namespace Z.Tool.#Name#;

class Entry : EntryEntry
{
    protected override long ExecuteMain()
    {
        Gen gen;
        gen = new Gen();
        gen.Init();
        long o;
        o = gen.Execute();
        return o;
    }

    [STAThread]
    static int Main(string[] arg)
    {
        return EntryEntry.Main(new Entry(), arg);
    }
}
